      def TestCase.inherited(subclass) # intercept test cases
        printf "\n%s\n", subclass.to_s.sub(/^Test/, '').sub(/Test$/, '')
        eval "          def \#{subclass}.method_added(id) \# intercept test methods\n            meth = id.to_s\n            printf(\"- %s\\n\", meth.sub(/^test_?/, '').gsub(/_/, ' ')) if meth =~ /^test/\n          end\n"
      end
